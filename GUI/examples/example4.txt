PROGRAM
  SELECT $v$ IN $V$
  $V₁ ← {v}$
  // V₂ изначально пусто
  $V₂ ← ∅$

  // Цикл по всем остальным вершинам u в графе (кроме v)
  FOR $u \in V \ {v}$ DO
    
    // Проверяем, является ли расстояние d(v, u) чётным
    IF $d(v, u) mod 2 = 0$ THEN // d(v, u) mod 2 = 0 означает, что остаток от деления на 2 равен 0
      // Если расстояние чётно, добавляем вершину u в первую долю (V₁)
      $V₁ ← V₁ ∪ {u}$ // V₁ ∪ {u} - объединение множеств V₁ и {u}
    ELSE
      // Если расстояние нечётно, добавляем вершину u во вторую долю (V₂)
      $V₂ ← V₂ ∪ {u}$ // V₂ ∪ {u} - объединение множеств V₂ и {u}
    END_IF
    
  END_FOR // Конец цикла по вершинам u

  // Возвращаем полученные доли (множества вершин)
  RETURN $(V₁, V₂)$
